cdol_ability_unlock_tooltip_effect = {
	if = {
		limit = {
			var:cdol_succubus_mastery_level = cdol_drain_life_succubus_mastery_level
		}
		custom_tooltip = cdol_drain_life_unlocked_tooltip
	}
	if = {
		limit = {
			var:cdol_succubus_mastery_level = cdol_fascinate_succubus_mastery_level
		}
		custom_tooltip = cdol_fascinate_unlocked_tooltip
	}
	if = {
		limit = {
			var:cdol_succubus_mastery_level = cdol_mesmerize_succubus_mastery_level
		}
		custom_tooltip = cdol_mesmerize_unlocked_tooltip
	}
	if = {
		limit = {
			var:cdol_succubus_mastery_level = cdol_succubus_conversion_succubus_mastery_level
		}
		custom_tooltip = cdol_succubus_conversion_unlocked_tooltip
	}
}
