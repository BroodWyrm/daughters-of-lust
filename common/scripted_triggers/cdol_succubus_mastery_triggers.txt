cdol_has_succubus_mastery_trigger = {
	custom_description = {
		text = "cdol_has_succubus_mastery_trigger_text"
		value = $VALUE$

		has_trait = cdol_succubus
		has_trait_xp = {
			trait = cdol_succubus
			value >= $VALUE$
		}
	}
}

cdol_succubus_mastery_equals_trigger = {
	has_trait = cdol_succubus
	has_trait_xp = {
		trait = cdol_succubus
		value = $VALUE$
    }
}
