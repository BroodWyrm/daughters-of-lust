cdol_is_succubus_trigger = {
	OR = {
		has_trait = cdol_succubus
		any_secret = {
			secret_type = cdol_secret_succubus
		}
	}
}

cdol_has_succubus_ancestry_trigger = {
	any_ancestor = {
		cdol_is_succubus_trigger = yes
	}
}

cdol_has_negative_attitude_towards_succubi_trigger = {
	cdol_is_succubus_trigger = no
	faith = { save_temporary_scope_as = my_faith }
	OR = {
		trait_is_shunned_in_faith_trigger = { TRAIT = witch FAITH = scope:my_faith GENDER_CHARACTER = $GENDER_CHARACTER$ }
		trait_is_criminal_in_faith_trigger = { TRAIT = witch FAITH = scope:my_faith GENDER_CHARACTER = $GENDER_CHARACTER$ }
	}
}
